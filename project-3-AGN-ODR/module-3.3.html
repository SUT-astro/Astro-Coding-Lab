Orthogonal distance regression (ODR) คืออะไร

ODR คือการวิเคราะห์หาค่าตัวแปรของสมการที่ดีที่สุด ด้วยการคำนวณหาตัวแปรของสมการที่ทำให้ระยะที่ลากตั้งฉากจากสมการไปยังจุดข้อมูลเฉลี่ยที่สั้นที่สุด ซึ่งจะแตกต่างกับ least square regression (LSR) ในจุดที่ LSR ใช้การพิจารณาระยะในแนวตั้ง หรือระยะตามแนวแกน y จากจุดข้อมูลไปยังสมการ ตามที่เห็นใน Figure… ซึ่งแสดงการพิจารณาหาตัวแปรของสมการที่ดีที่สุดในทั้ง 2 algorithm

ข้อดีของการทำ ODR คือ เราสามารถพิจารณาค่า error ของข้อมูลทั้งแกน x และแกน y ได้พร้อมกันในขณะ fit กราฟ ต่างจาก LSR ที่พิจารณาเพียงค่า error ของข้อมูลในแกน y เท่านั้น

โดยในแบบฝึกหัดนี้ เราจะใช้ฟังชั่น scipy.odr จาก scipy ในการทำ ODR ซึ่งมีตัวอย่างการใช้งานดังนี้

  1. import module ที่เราจะใช้
    2.               
import numpy as np
import pandas as pd
import matplotlib.pyplot as plt

from scipy.odr import ODR, Model, RealData
  2. สร้าง array เก็บข้อมูลที่จะทำการ fit กราฟ
    3.               
# Data, t (time) on x-axis and v (velocity) on y-axis
t = np.arange(0, 1, 0.1)
t_err = np.zeros(len(t)) + 0.02

v = np.array([0.5, 1.3, 2.7, 3.4, 4.3, 5.2, 6.3, 7.1, 8.6, 9.4])
v_err = np.array([0.05, 0.1, 0.4, 0.1, 0.2, 0.2, 0.1, 0.4, 0.3, 0.4])

# Create a dataframe
table_column = {'t (s)': t, 't_error (s)': t_err, 'v (cm/s)': v, 'v_error (cm/s)': v_err}
df = pd.DataFrame(data = table_column)
df.head(3)
  3. สร้างฟังชั่นของสมการที่เราจะใช้ในการ fit กับข้อมูล ซึ่งในที่นี้เป็น function_linear()
    4.               
def function_linear(B, x):
    '''Linear eqution: y = b + m*x'''
    # B is a vector of the parameters, which are B[0] for b or intercept in the equation, and B[1] for m or slope in the equation.
    # x is an array of the current x values.
    # x is in the same format as the x passed to Data or RealData.

    # Return an array in the same format as y passed to Data or RealData.
    return B[0] + (B[1] * x)
  4. นิยามโมเดลของสมการที่เราจะใช้ด้วย Model() และนิยามข้อมูลที่เราจะใช้ด้วย RealData()
    5.               
# Define Model and Data
linear_model = Model(function_linear)
data = RealData(x = t, y = v , sx = t_err, sy = v_err)
  5. ตั้งค่า algorithm ของการ fit กราฟด้วย ODR()
    6.               
# Configure the Orthogonal Distance Regression
odr = ODR(data, linear_model, [1, 1])
odr.set_job(fit_type = 0)       # 0: explicit ODR, 2: least square method (use for verification)
  6. ทำการ fit กราฟ
    7.               
# Perform ODR fitting
output = odr.run()
  7. แสดงผลลัพธ์จากการ fit กราฟ
    8.               
# The prediction result of parameters in linear equation
constant, slope = output.beta
constant_err, slope_err = output.sd_beta    # The error (or variance) of predicted parameters
print('Intercept is ', constant, ', error = ', constant_err)
print('Slope is ', slope, ', error = ', slope_err)
  8. นำผลลัพธ์ที่ได้มา plot กราฟเส้นตรง
    9.               
# Find a linear equation from predicted result
x_predict = np.linspace(0, t[-1], 1000)
y_predict = function_linear(output.beta, x_predict)
y_min = function_linear(B = [constant + constant_err, slope - slope_err], x = x_predict)
y_max = function_linear(B = [constant - constant_err, slope + slope_err], x = x_predict)

# Graph plotting
plt.rcParams.update({'font.size': 14})
fig = plt.figure(figsize = (5, 8))
ax = fig.add_subplot(111)

ax.plot(t, v, 'ok', markersize = 8, label = 'Data')
ax.errorbar(t, v, xerr = t_err, yerr = v_err, fmt = 'none', ecolor = 'orange', capsize = 3.0, label = 'Error of Data')

ax.plot(x_predict, y_predict, '-b', label = 'Prediction by ODR')
ax.plot(x_predict, y_min, '--g')
ax.plot(x_predict, y_max, '--r')

ax.set_title('Figure of velocity at different time')
ax.set_xlabel('t (s)')
ax.set_ylabel('v (cm/s)')
ax.text(0.35, 0.2, r'Slope = {0:.1f} $\pm$ {1:.1f} cm/s$^2$'.format(slope, slope_err), transform=ax.transAxes)
ax.text(0.2, 0.1, r'Intercept = {0:.2f} $\pm$ {1:.2f} cm/s'.format(constant, constant_err), transform=ax.transAxes)
ax.legend(loc = 'best')
ax.grid()
plt.show()
